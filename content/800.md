    [code language="php"]

    <?php

    function array_sum_values(array $input, $key) {

       $sum = 0;

       array_walk($input, function($item, $index, $params) {

          if (!empty($item[$params[1]]))

             $params[0] += $item[$params[1]];

       }, array(&$sum, $key));

       return $sum;

    }
